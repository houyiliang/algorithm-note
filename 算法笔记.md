```python
def hanoi(n , a , b , c):    #a,b,c代表从a移动到c中间经过b
    if(n>0):
        hanoi(n-1 , a ,c ,b)
        print("moving from %s to %s" %(a , c))
        hanoi(n-1 , b , a ,c)
```

## 3、二分法查找（局限性必须是有序数组）

```python
def binary_search(li ,val):   #li代表列表  val代表要查找的值
    left = 0
    right = len(li)-1
    while right > left:
        mid = (right + left)//2
        if  li[mid] == val:
            return mid
        elif li[mid] > val:
            right = mid-1
        else:
            left = mid + 1
    else:
        return None

li = list(range(100))   				#range(stop)  range(start, stop ，step)默认start为0，step为1
print(binary_search(li,42))
```

```c++
#include<iostream>
using namespace std;

int binary_search(int array[],int val)
{
	int left = 0;
	int right =  sizeof(array)/4;
	while (left <= right)
	{
		int mid = (right + left) / 2;
		if (array[mid] == val)
		{
			return mid;
			break;
		}
		else if (array[mid] > val)
		{
			right = mid - 1;
		}
		else
		{
			left = mid + 1;
		}
	}
	if (right < left)
	{
		return -1;
	}
	
}
```

## 4、冒泡排序

```
import random

def bubble_sort(li):
    for i in range(len(li)-1):           #第i轮          range(start, stop[, step])
        flage = False					#冒泡优化
        for j in range(len(li)-i-1):
            if li[j]>li[j+1]:
                li[j],li[j+1]=li[j+1],li[j]
                flage = True
        if not flage:					#当后面顺序排好时不在重复循环节省时间
            return



li = [random.randint(0,100) for i in range(100)]
print(li[1] , li[6])
bubble_sort(li)
print(li)
```

## 5、选择排序

这种需要创建一个新的列表占用的空间大，时间复杂度为O(n的平方)

```python
import random

def select_sort_simple(li):
    li_new = []
    for i in range(len(li)):
        min_val = min(li)       #min内置函数寻找li列表中最小的元素
        li_new.append(min_val)
        li.remove(min_val)		#remove删除列表中的元素
    return li_new

li = [random.randint(0,20) for i in range(20)]
print(select_sort_simple(li))

```

这种情况无需创建新列表，空间复杂度减小，时间复杂度不变

```python
import random
def select_sort(li):
    for i in range(len(li)-1):          # i是第几趟
        min_loc = i                     #设i为最小值
        for j in range(i+1,len(li)):    #将i后的每一个数与min相比，若小于则更新min
            if li[j] < li[min_loc]:
                min_loc = j
        li[i], li[min_loc] = li[min_loc] , li[i]        #交换i和min_loc位置永远保证小的在前面



li = [random.randint(0,50) for i in range(20)]
select_sort(li)
print(li)
```

```C++
#include<iostream>
using namespace std;
#include<ctime>

void select_sort(int a[])
{
	for (int i = 0; i < 20; i++)
	{
		int min_loc = i;
		for (int j = i; j < 20; j++)
		{
			if (a[j] < a[min_loc])
			{
				min_loc = j;
			}
		}
		int tem = a[i];
		a[i] = a[min_loc];
		a[min_loc] = tem;
	}
}

int main()
{
	srand((unsigned int)time(NULL));
	int a[20];
	for (int i = 0; i < 20; i++)
	{
		
		a[i] = rand() % 20;
	}
	select_sort(a);
	for (int i = 0; i < 20; i++)
	{
		cout << a[i] << " ";
	}
	cout << endl;
	system("pause");
	return 0;
}
```

```python
import random
def insert_sort(li):
    for i in range(len(li)):
        tem = li[i]
        j = i - 1
        while j >= 0 and li[j] > tem:
            li[j+1] = li [j]
            j -= 1
        li[j+1] = tem
```

```
#include<iostream>

using namespace std;

void insert_sort(int *arr)
{
	
	for (int i = 0; i < 20; i++)	#这里i的上限不能用sizeof(arr) 子函数中等于一个数的字节
	{
		int temp = arr[i];
		int j = i - 1;
		while (j >= 0 && arr[j] > temp)
		{
			arr[j + 1] = arr[j];
			j--;
		}
		arr[j + 1] = temp;
	}
}
```
